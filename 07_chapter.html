6. Technical Implementation Guide

Implementing AG-UI and A2UI protocols requires careful consideration of architecture, tooling, and integration patterns. This chapter provides comprehensive guidance for developers building agent-driven applications.

Getting Started

The fastest path to experimenting with A2UI concepts is through the A2UI Widget Builder at https://a2ui-composer.ag-ui.com/gallery. This interactive tool, powered by CopilotKit, allows developers to explore component generation without full integration setup. Additional resources:

  • Official Documentation: https://a2ui.org/ — Comprehensive specs, quickstart guides, and demos
  • GitHub Repository: https://github.com/google/A2UI — Source code and examples
  • CopilotKit Docs: https://docs.copilotkit.ai/ — AG-UI integration guides

Complete React Implementation

// Full-featured React component implementation import React, { useState, useCallback, useEffect } from 'react'; import {   CopilotProvider,   useStream,   useCopilotAction,   useCopilotState } from '@copilotkit/react'; import { A2UIRenderer, A2UIComponentCatalog } from '@google/a2ui-react';  interface AgentInterfaceProps {   agentConfig: AGUIConfig;   componentCatalog: A2UIComponentCatalog;   onError?: (error: Error) => void;   className?: string; }  export function AgentInterface({   agentConfig,   componentCatalog,   onError,   className }: AgentInterfaceProps) {   const [messages, setMessages] = useState<Message[]>([]);   const [uiComponents, setUIComponents] = useState<A2UIComponent[]>([]);   const [connectionState, setConnectionState] = useState<'connecting' | 'connected' | 'error'>('connecting');    const {     isLoading,     sendMessage,     streamingMessage,     error   } = useStream({     config: agentConfig,     onGenerativeUI: handleGenerativeUI,     onMessage: handleMessage,     onError: handleError,     onConnectionStateChange: setConnectionState   });    const handleGenerativeUI = useCallback((payload: A2UIPayload) => {     console.log('Processing A2UI payload:', payload);          switch (payload.type) {       case 'surface_initialize':         setUIComponents([]);         break;                case 'component_add':         setUIComponents(prev => [...prev, payload.component]);         break;                case 'component_update':         setUIComponents(prev =>           prev.map(comp =>             comp.id === payload.component_id               ? { ...comp, properties: { ...comp.properties, ...payload.updates.properties } }               : comp           )         );         break;                case 'component_remove':         setUIComponents(prev =>           prev.filter(comp => comp.id !== payload.component_id)         );         break;                default:         console.warn('Unknown A2UI payload type:', payload.type);     }   }, []);    const handleA2UIEvent = useCallback((event: A2UIEvent) => {     console.log('A2UI event:', event);          sendMessage({       type: 'USER_INTERACTION',       payload: {         surface_id: event.surfaceId,         component_id: event.componentId,         event_type: event.type,         data: event.data,         timestamp: new Date().toISOString(),         user_context: {           viewport: {             width: window.innerWidth,             height: window.innerHeight           },           device_type: /Mobile/.test(navigator.userAgent) ? 'mobile' : 'desktop'         }       }     });   }, [sendMessage]);    const handleMessage = useCallback((message: Message) => {     setMessages(prev => [...prev, message]);   }, []);    const handleError = useCallback((error: Error) => {     console.error('AG-UI Error:', error);     onError?.(error);     setConnectionState('error');   }, [onError]);    const handleRetryConnection = useCallback(() => {     setConnectionState('connecting');     // Implement reconnection logic   }, []);    useEffect(() => {     // Setup cleanup on unmount     return () => {       setMessages([]);       setUIComponents([]);     };   }, []);    return (     <CopilotProvider config={agentConfig}>       <div className={`agent-interface ${className || ''}`}>         <ConnectionStatus            state={connectionState}            onRetry={handleRetryConnection}          />                  <div className="conversation-area">           <MessageList              messages={messages}              isLoading={isLoading}             error={error}           />                      {streamingMessage && (             <StreamingMessage                content={streamingMessage}               showCursor={isLoading}              />           )}         </div>                  <div className="generative-ui-area">           {uiComponents.length > 0 && (             <A2UIRenderer               components={uiComponents}               catalog={componentCatalog}               onEvent={handleA2UIEvent}               theme="default"               accessibility={{                 announceChanges: true,                 focusManagement: 'auto'               }}             />           )}         </div>                  <MessageInput           onSend={sendMessage}           disabled={isLoading || connectionState !== 'connected'}           placeholder={             connectionState === 'connected'                ? "Type your message..."               : "Connecting..."           }         />       </div>     </CopilotProvider>   ); }

// LangGraph agent with AG-UI + A2UI integration from langchain_core.messages import HumanMessage, SystemMessage from langgraph import StateGraph, END from copilotkit.ag_ui import AGUIAdapter, A2UIGenerator import asyncio import json  class RestaurantBookingAgent:     def __init__(self):         self.agui_adapter = AGUIAdapter(             endpoint="wss://localhost:8000/agui",             auth_token=os.getenv("AGUI_AUTH_TOKEN")         )         self.a2ui_generator = A2UIGenerator()         self.llm = ChatOpenAI(model="gpt-4", temperature=0.1)              async def process_booking_request(self, state):         """Main booking processing workflow"""         user_message = state['messages'][-1].content                  # Initialize booking interface         await self.initialize_booking_surface()                  # Extract booking requirements         requirements = await self.extract_booking_requirements(user_message)                  # Generate appropriate form based on requirements         booking_form = await self.generate_booking_form(requirements)                  # Send form to user via AG-UI         await self.agui_adapter.send_event({             "type": "GENERATIVE_UI_UPDATE",             "payload": booking_form         })                  return state          async def initialize_booking_surface(self):         """Initialize the booking UI surface"""         surface_config = self.a2ui_generator.create_surface({             "surface_id": f"booking_{uuid.uuid4()}",             "title": "Restaurant Reservation",             "layout": {                 "type": "card_layout",                 "max_width": "600px",                 "padding": "medium"             },             "accessibility": {                 "aria_label": "Restaurant booking form",                 "focus_management": "auto"             }         })                  await self.agui_adapter.send_event({             "type": "GENERATIVE_UI_UPDATE",              "payload": surface_config         })          async def generate_booking_form(self, requirements):         """Generate booking form based on extracted requirements"""                  # Use LLM to determine appropriate form fields         form_prompt = f"""         Generate a restaurant booking form based on these requirements:         {json.dumps(requirements, indent=2)}                  Return A2UI JSON format with appropriate form fields.         Include validation rules and helpful placeholder text.         """                  response = await self.llm.ainvoke([             SystemMessage(content=self.get_a2ui_system_prompt()),             HumanMessage(content=form_prompt)         ])                  try:             form_config = json.loads(response.content)             return self.a2ui_generator.validate_and_enhance(form_config)         except json.JSONDecodeError:             # Fallback to default form             return self.generate_default_booking_form()          async def handle_form_submission(self, state):         """Process submitted booking form"""         form_data = state.get('form_submission')                  # Validate form data         validation_result = await self.validate_booking_data(form_data)                  if not validation_result['is_valid']:             # Show validation errors             error_ui = self.generate_error_display(validation_result['errors'])             await self.agui_adapter.send_event({                 "type": "GENERATIVE_UI_UPDATE",                 "payload": error_ui             })             return state                  # Process booking with restaurant API         booking_result = await self.process_restaurant_booking(form_data)                  # Generate confirmation UI         confirmation_ui = await self.generate_confirmation_ui(booking_result)         await self.agui_adapter.send_event({             "type": "GENERATIVE_UI_UPDATE",             "payload": confirmation_ui         })                  return state          def get_a2ui_system_prompt(self):         return """         You are an expert at generating A2UI form configurations.         Always respond with valid JSON following the A2UI specification.                  Guidelines:         - Use appropriate input types for different data         - Include helpful validation rules         - Add clear labels and placeholder text         - Consider accessibility requirements         - Keep forms concise but comprehensive                  Available component types:         - text_input, email_input, phone_input         - number_input, date_input, time_input         - select, radio_group, checkbox_group         - button, submit_button         """  # Usage example async def main():     agent = RestaurantBookingAgent()          # Process incoming message     state = {         'messages': [             HumanMessage(content="I'd like to book a table for 4 people tomorrow evening")         ]     }          result = await agent.process_booking_request(state)     print("Booking agent processed request:", result)
